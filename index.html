# Setup git --version # Check Git version git config --global user.name "Name" # Set name git config --global user.email "Email" # Set email # Repository git init # Initialize repo git clone # Clone repo # Staging & Committing git add # Stage file git add . # Stage all files git commit -m "Message" # Commit changes git commit -am "Message" # Stage & commit tracked # Status & Logs git status # Show status git log # Show commits git log --oneline # Compact log # Branching git branch # List branches git branch # Create branch git checkout # Switch branch git checkout -b # Create & switch branch git merge # Merge branch # Remote git remote -v # List remotes git remote add # Add remote git pull # Fetch & merge remote changes git push # Push changes git push -u origin # Push and set upstream # Undoing git reset # Unstage file git reset --hard # Reset to last commit git checkout -- # Discard file changes git revert # Undo commit by new commit # Stash git stash # Save changes git stash list # List stashes git stash apply # Apply stash git stash drop # Remove stash # Misc git diff # Show differences git clean -f # Remove untracked files git tag # Create tag
